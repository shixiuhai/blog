class MyThread(Thread):
    def __init__(self, func, args):
        self.lock = threading.RLock()
        super(MyThread, self).__init__()
        self.func = func
        self.args = args

    def run(self):
        # 锁上
        self.lock.acquire()  
        self.result = self.func(*self.args)
        # 解锁
        self.lock.release()

    def get_result(self):
        try:
            return self.result
        except Exception:
            return None
