from logging import error

import cv2

from jobList import Jobs
from kafka import KafkaProducer
def detect(source,kafkaProducer,id,weights, monitorType):
    count=0
    notice = True
    # 加载模型
    model = torch.hub.load('yolov6', 'custom', path='%s'%weights, source='local')
    # 读取视频
    # 创建视频对象
    cap = cv2.VideoCapture(source)
    # 视频宽
    width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    rtmp_fps=int(cap.get(cv2.CAP_PROP_FPS)%100)
    rtmp_url='rtmp://10.8.0.42:1935/live/1_%s_%s'%(monitorType,id)
    """推流初始化参数"""
    command = ['ffmpeg',
                        '-y',
                        '-f', 'rawvideo',
                        '-vcodec', 'rawvideo',
                        '-pix_fmt', 'bgr24',
                        '-s', "&#123;&#125;x&#123;&#125;".format(width, height),
                        '-r', str(rtmp_fps),
                        '-i', '-',
                        '-c:v', 'libx264',
                        '-pix_fmt', 'yuv420p',
                        '-preset', 'ultrafast',
                        '-f', 'flv',
                        rtmp_url]
    pipe = subprocess.Popen(command, shell=False,  stdin=subprocess.PIPE)
    try:
        while cap.isOpened():
            #if id not in Jobs._job or id in Jobs._restart_job:
                #break
            ret, frame = cap.read()
            # cv2.imwrite('./1.jpg',frame)
            # status 状态判断
            #if (id not in Jobs._job) or id in Jobs._restart_job:
            # time.sleep(0.5)
            if not ret:
                # 调用kafka发送错误信息
                # kafka_producer(&#123;'status':'0'&#125;,server_ip,(self.type+self.id))
                print(source + "打开视频源失败")
                break
            
            # 检测
            img = Image.fromarray(cv2.cvtColor(frame,cv2.COLOR_BGR2RGB))  
            # print(img)
            # 调用模型
            results = model(img, size=640)
            # 获取到图片画框json数据
            results_objectrecognition = json.loads(results.pandas().xyxy[0].to_json(orient="records"))
            truck_num=len(results_objectrecognition)
            #print("-------------------")
            #print(Jobs._job)
            #print("-------------------")
            if notice:
                if id in Jobs._restart_job:
                    Jobs._restart_job.remove(id)
                kafkaProducer.send('checkStatus',json.dumps(&#123;'id':id,'monitorType':monitorType,'status':1&#125;).encode())
                notice = False
                #print("成功发送启动成功消息")
            if id not in Jobs._job:
                pipe.terminate()
                raise StopIteration
                return False
            if id in Jobs._restart_job:
                pipe.terminate()
                raise StopIteration
                return False
            for i in range(len(results_objectrecognition)):
                # 这里面有多个
                # 取出每个对象返回的 json 结果
                name = results_objectrecognition[i]['name']
                # confidence_result是浮点型数据
                confidence_result = results_objectrecognition[i]['confidence']
                xmin = results_objectrecognition[i]['xmin']
                ymin = results_objectrecognition[i]['ymin']
                xmax = results_objectrecognition[i]['xmax']
                ymax = results_objectrecognition[i]['ymax']
                # 把这四个做标转换为x,y,width,height
                x = int(xmin)
                y =  int(ymin)
                w =  int(xmax-xmin+1)
                h =  int(ymax-ymin+1)
                #print('置信度是%s'%confidence_result)
                if confidence_result >= float(0.70):
                    """
                    将目标图片写入到存放违规图片的文件夹
                    """
                    # 画框
                    frame= cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 0, 255), thickness=3)
                    # if name=="nomask":
                    #     cv2.putText(frame,"nomask %.2s%%"%(confidence_result*100),(x+w+5,y-5),cv2.FONT_HERSHEY_SIMPLEX,1,(255,0,0),2)
                    # elif name=="smoke":
                    #     cv2.putText(frame,"smoke %.2s%%"%(confidence_result*100),(x+w+5,y-5),cv2.FONT_HERSHEY_SIMPLEX,1,(255,0,0),2)
                    cv2.putText(frame,"%s %.2s%%"%(name,confidence_result*100),(x+w+5,y-5),cv2.FONT_HERSHEY_SIMPLEX,1,(255,0,0),2)
                    # 添加中文显示
                    # image,strs,local(图片的坐标),sizes,color 
                    # frame = change_cv2_draw(frame,"%s %.2s%%"%('吸烟',confidence_result*100),(x+w+5,y-5),30,(255,0,0))
                    # 将画好的图片赋值给self.picture 这个全局变量
                    
                    # cv2.imwrite('target_images/2.jpg',self.picture)
                    # cv2.imwrite("%s/'%s'.jpg"%('target_images',time.strftime('%Y-%m-%d-%H:%M:%S',time.localtime(time.time()))),self.picture)
                    count = count+1
                    if count == 100:
                        count=0
                        """保存图片"""
                        date=time.strftime("%Y-%m-%d",time.localtime())
                        if not os.path.exists("/home/wenhua/python/static/img/1,%s/%s/"%(monitorType,date)):
                            os.makedirs("/home/wenhua/python/static/img/1,%s/%s/"%(monitorType,date))
                        # 获取当前时间的字符串
                        currentTime = time.time()
                        cv2.imwrite("/home/wenhua/python/static/img/1,%s/%s/%s%d.png"%(monitorType,date,id,currentTime), frame)
                        # print("截图完成")
                        warnResult=Jobs._job_param[id]
                        kafkaProducer.send('checkWarning',json.dumps(&#123;'taskId':id,'companyId':warnResult['companyId'],'type':'1,%s'%(monitorType),'count':truck_num,'imgUrl':'1,%s/%s/%s%d.png'%(monitorType,date,id,currentTime)&#125;).encode())
            #print('在推流中')
            pipe.stdin.write(frame.tostring())
            # self.draw_picture(0.1)
            # cv2.imshow('img'+path,self.picture)
            cv2.waitKey(1)
    except:
        Jobs._restart_job.append(id)
        # 关闭通道
        pipe.stdin.close()
        print(source+'推流结束')
        #pipe.terminate()
        #raise StopIteration

# 定义一个添加中文方法
def change_cv2_draw(image,strs,local,sizes,colour):
    cv2img = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    pilimg = Image.fromarray(cv2img)
    draw = ImageDraw.Draw(pilimg)  # 图片上打印
    font = ImageFont.truetype("SIMYOU.TTF",sizes, encoding="utf-8")
    draw.text(local, strs, colour, font=font)
    image = cv2.cvtColor(np.array(pilimg), cv2.COLOR_RGB2BGR)
    return image
