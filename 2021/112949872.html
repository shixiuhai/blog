linux 使用nohup创建后台

# 将 check_application.py放入后台运行并生成日志，日志目录在info.log
# 2>&1 的意思就是将标准错误重定向到标准输出
nohup python3 check_application.py >log/info.log 2>&1 &

linux vim使用多窗口

水平窗口的开启使用:sp命令即可 # （1）Ctrl+w+k 上一个窗口   Ctrl+w+ j  下一个窗口 （2）Ctrl+w+上下方向键
垂直窗口的开启使用:vs命令即可 # （1）Ctrl+w+h 当前窗口左边的一个窗口   Ctrl+w+ l  当前窗口右边的一个窗口 （2）Ctrl+w+左右方向键

linux 获取某个进程的pid号

ps -ef | grep check_application.py  | grep -v 'grep'| awk '&#123;print $2&#125;' # check_application.py 是运行程序的名字
ps -ef | grep traffic_application.py | grep -v 'grep'| awk '&#123;print $2&#125;'
ps -ef | grep truck_application.py | grep -v 'grep'| awk '&#123;print $2&#125;'
ps -ef | grep 'SCREEN -S test'  | grep -v 'grep'| awk '&#123;print $2&#125;' # 获取screen 某个虚拟终端的id
## 某个脚本案例
#!/bin/bash
# 获取pid程序运行的pid号
pid1=$(ps -ef | grep check_application.py  | grep -v 'grep'| awk '&#123;print $2&#125;')
pid2=$(ps -ef | grep traffic_application.py  | grep -v 'grep'| awk '&#123;print $2&#125;')
pid3=$(ps -ef | grep truck_application.py  | grep -v 'grep'| awk '&#123;print $2&#125;')
# 杀死已经运行的程序
kill -9 $pid1
kill -9 $pid2
kill -9 $pid3
# 进入虚拟环境
source env/bin/activate
# 后台运行程序
nohup python3 traffic_application.py >log/trafficInfo.log.log 2>&1 &
nohup python3 check_application.py >log/info.log 2>&1 &
nohup python3 truck_application.py >log/truckInfo.log 2>&1 &
# 退出虚拟环境
deactivate
echo "check_application.py traffic_application.py truck_application.py重启成功"

linux设置环境变量

# bash 终端的情况下
## 1. 对所有用户有效
vi /etc/profile
## 所有的引号可以不添加PATH必须大写
export PATH="$PATH:/usr/abc/def/"
source /etc/profile
## 2. 对当前用户有效
vi /root/.bashrc
export PATH="$PATH:/usr/abc/def/"
source /root/.bashrc

## 3. 添加多个环境变量
# 比如添加搜索路径/home/cc/path1和/home/cc/path2,路径到PATH中,采用:来分隔,冒号左右不需要空格:
export PATH="$PATH:/home/cc/path1:/home/cc/path2"

screen 脚本使用

## screen 配置开机自启动脚本
screen_name="my_screen" 
cmd="python test.py";  
screen -dmS $screen_name # 创建名字
# 方式一
screen -x -S $screen_name -p 0 -X stuff "$cmd\n" # 运行命令 
# 方式二 回车符可以在下一行传入
screen -x -S $screen_name -p 0 -X stuff "$cmd" # 运行命令不添加回车符号
screen -x -S $screen_name -p 0 -X stuff '\n' # 添加回车符号

linux开机自启动两种方法

## 1. 使用update-rc.d 
# 创建shell脚本并移动到/etc/init.d里面
cd /etc/init.d
# 配置开机启动
update-rc.d name defaults # 或者 update-rc.d name start 20 2 3 4 5 . stop 20 0 1 6 .


## 2. 使用systemd
# 进入到system下
/usr/lib/systemd/system/
# 配置一个service文件
vim test.service
# 内容如下例子
[Unit]
Description=Frp Client Service
After=network.target
[Service]
Type=simple
User=nobody
Restart=on-failure
RestartSec=5s
ExecStart=/usr/bin/frpc -c /etc/frp/frpc.ini
ExecReload=/usr/bin/frpc reload -c /etc/frp/frpc.ini
LimitNOFILE=1048576
[Install]
WantedBy=multi-user.target

# 启动配置使用systemctl命令即可
systemctl start test
